Educational Outcomes Markup

Schema

Note: Links in this section will work after the RDF is checked in to upload our schema to schema.googleapis.com.
  • New GoogleAPI property educationalOutcomes for schema.org type CollegeOrUniversity
  • with expected value of the new GoogleAPI type EducationalOutcomes
  • New GoogleAPI types EducationalFacet and EducationalOutcome

Template

    {
      "@context": "https://schema.org",
      "@type": "CollegeOrUniversity",

      // Information about the college.
      // Fields for college name, address, any other globally unique identifiers.

      "educationalOutcomes": {  // New property of CollegeOrUniversity.
        "@context": "https://schema.googleapis.com",
        "@type": "EducationalOutcomes",  // New type.
        // TODO "numStudents": "$num",

        // -------------------------------------------------------------------
        // Facets fieldOfStudyFacet, companyFacet, functionFacet.
        //
        // Optional facet for program-level outcomes.
        "fieldOfStudyFacet": {
          "@type": "EducationalFacet",
          "name": "$field_name",
          "identifier": "$id"
        },
        // Optional facet for company-level outcomes.
        "companyFacet": {
          "@type": "EducationalFacet",
          "name": "$company_name",
          "sameAs": "$url",
          "identifier": "$id"
        },
        // Optional facet for function-level outcomes.
        "functionFacet": {
          "@type": "EducationalFacet",
          "name": "$function_name",
          "sameAs": "$url",
          "identifier": "$id",
        },

        // -------------------------------------------------------------------
        // Outcomes.

        "fieldsOfStudy": [  // "What they studied", or "facetFieldOfStudy"
          {
            "@type": "EducationalOutcome",
            "name": "$field_name",
            "identifier": "$id",
            "numStudents": "$num",
            "thumbnail": "$logo_url"
          }
          // More fields of study..
        ],

        "companies": [  // "Where they work", or "facetCurrentCompany"
          {
            "@type": "EducationalOutcome",
            "name": "$company_name",
            "sameAs": "$url",
            "identifier": "$id",
            "numStudents": "$num",
            "thumbnail": "$logo_url"
          }
          // More companies..
        ],

        "functions": [  // "What they do", or "facetCurrentFunction"
          {
            "@type": "EducationalOutcome",
            "name": "$function_name",
            "sameAs": "$url",
            "identifier": "$id",
            "numStudents": "$num",
            "thumbnail": "$logo_url"
          },
          // More functions..
        ]
      }
    }
